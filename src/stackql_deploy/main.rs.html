3//! # StackQL Deploy - Main Entry Point
4//!
5//! This is the main entry point for the StackQL Deploy application.
6//! It initializes the CLI, configures global settings, and handles user commands (e.g., `build`, `teardown`, `test`, `info`, `shell`, etc.).
7//!
8//! ## Global Arguments
9//!
10//! These arguments can be specified for **any command**.
11//!
12//! - `--server`, `-h` - The server host to connect to (default: `localhost`).
13//! - `--port`, `-p` - The server port to connect to (default: `5444`).
14//! - `--log-level` - The logging level (default: `info`). Possible values: `error`, `warn`, `info`, `debug`, `trace`.
15//!
16//! ## Example Usage
17//! ```bash
18//! ./stackql-deploy --server myserver.com --port 1234 build
19//! ./stackql-deploy shell -h localhost -p 5444
20//! ./stackql-deploy info
21//! ```
22//!
23//! For detailed help, use `--help` or `-h` flags.
24

34use std::process;
35
36use clap::{Arg, ArgAction, Command};
37
38use error::{get_binary_path_with_error, AppError};
39use log::{debug, error, info};
40
41use crate::app::{
42    APP_AUTHOR, APP_DESCRIPTION, APP_NAME, APP_VERSION, DEFAULT_LOG_LEVEL, DEFAULT_SERVER_HOST,
43    DEFAULT_SERVER_PORT, DEFAULT_SERVER_PORT_STR, EXEMPT_COMMANDS, LOG_LEVELS,
44};
45use crate::utils::logging::initialize_logger;
46
47/// Main function that initializes the CLI and handles command execution.
48fn main() {
49    let matches = Command::new(APP_NAME)
50        .version(APP_VERSION)
51        .author(APP_AUTHOR)
52        .about(APP_DESCRIPTION)
53        // ====================
54        // Global Flags
55        // ====================
56        .arg(
57            Arg::new("server")
58                .long("server")
59                .alias("host")
60                .short('H')
61                .help("StackQL server host to connect to")
62                .global(true)
63                .default_value(DEFAULT_SERVER_HOST)
64                .action(ArgAction::Set),
65        )
66        .arg(
67            Arg::new("port")
68                .short('p')
69                .long("port")
70                .help("StackQL server port to connect to")
71                .value_parser(clap::value_parser!(u16).range(1024..=65535))
72                .global(true)
73                .default_value(DEFAULT_SERVER_PORT_STR)
74                .action(ArgAction::Set),
75        )
76        .arg(
77            Arg::new("log-level")
78                .long("log-level")
79                .help("Set the logging level")
80                .global(true)
81                .value_parser(clap::builder::PossibleValuesParser::new(LOG_LEVELS))
82                .ignore_case(true)
83                .default_value(DEFAULT_LOG_LEVEL)
84                .action(ArgAction::Set),
85        )
86        .subcommand_required(true)
87        .arg_required_else_help(true)
88        // ====================
89        // Subcommand Definitions
90        // ====================
91        .subcommand(commands::build::command())
92        .subcommand(commands::teardown::command())
93        .subcommand(commands::test::command())
94        .subcommand(commands::info::command())
95        .subcommand(commands::shell::command())
96        .subcommand(commands::upgrade::command())
97        .subcommand(commands::init::command())
98        .subcommand(commands::start_server::command())
99        .subcommand(commands::stop_server::command())
100        .subcommand(commands::plan::command())
101        .get_matches();
102
103    // ====================
104    // Initialize Logger
105    // ====================
106    let log_level = matches.get_one::<String>("log-level").unwrap();
107    initialize_logger(log_level);
108
109    debug!("Logger initialized with level: {}", log_level);
110
111    // Get the server and port values from command-line arguments
112    let server_host = matches
113        .get_one::<String>("server")
114        .unwrap_or(&DEFAULT_SERVER_HOST.to_string())
115        .clone();
116
117    let server_port = *matches
118        .get_one::<u16>("port")
119        .unwrap_or(&DEFAULT_SERVER_PORT);
120
121    debug!("Server Host: {}", server_host);
122    debug!("Server Port: {}", server_port);
123
124    // Initialize the global values
125    globals::init_globals(server_host, server_port);
126
127    // Check for binary existence except for exempt commands
128    if !EXEMPT_COMMANDS.contains(&matches.subcommand_name().unwrap_or("")) {
129        match get_binary_path_with_error() {
130            Ok(path) => debug!("StackQL binary found at: {:?}", path),
131            Err(_e) => {
132                info!("StackQL binary not found. Downloading the latest version...");
133                commands::upgrade::execute();
134
135                // Re-check for binary existence after upgrade attempt
136                if let Err(AppError::BinaryNotFound) = get_binary_path_with_error() {
137                    error!("Failed to download StackQL binary. Please try again or check your network connection.");
138                    process::exit(1);
139                }
140            }
141        }
142    }
143
144    // ====================
145    // Command Execution
146    // ====================
147    match matches.subcommand() {
148        Some(("build", sub_matches)) => commands::build::execute(sub_matches),
149        Some(("test", sub_matches)) => commands::test::execute(sub_matches),
150        Some(("plan", sub_matches)) => commands::plan::execute(sub_matches),
151        Some(("teardown", sub_matches)) => commands::teardown::execute(sub_matches),
152        Some(("info", _)) => commands::info::execute(),
153        Some(("shell", sub_matches)) => commands::shell::execute(sub_matches),
154        Some(("upgrade", _)) => commands::upgrade::execute(),
155        Some(("init", sub_matches)) => commands::init::execute(sub_matches),
156        Some(("start-server", sub_matches)) => commands::start_server::execute(sub_matches),
157        Some(("stop-server", sub_matches)) => commands::stop_server::execute(sub_matches),
158        _ => {
159            print_error!("Unknown command. Use --help for usage.");
160            process::exit(1);
161        }
162    }
163}
