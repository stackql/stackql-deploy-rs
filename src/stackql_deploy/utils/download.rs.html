3//! # Download Utility Module
4//!
5//! This module provides functions for downloading, extracting, and setting up the StackQL binary.
6//! It supports various platforms including Linux, Windows, and macOS, handling differences in
7//! extraction methods and permissions.
8//!
9//! ## Features
10//! - Downloads the StackQL binary from a predefined URL.
11//! - Supports progress tracking during download.
12//! - Extracts the binary on various platforms (Windows, Linux, macOS).
13//! - Sets executable permissions on Unix-like systems.
14//!
15//! ## Example Usage
16//! ```rust
17//! use crate::utils::download::download_binary;
18//!
19//! match download_binary() {
20//!     Ok(path) => println!("Binary downloaded to: {}", path.display()),
21//!     Err(e) => eprintln!("Failed to download binary: {}", e),
22//! }
23//! ```

39/// Retrieves the URL for downloading the StackQL binary.
40pub fn get_download_url() -> Result<String, AppError> {
41    Ok(STACKQL_DOWNLOAD_URL.to_string())
42}

44/// Downloads the StackQL binary and extracts it to the current directory.
45///
46/// This function downloads the StackQL binary from a URL and unzips it if necessary.
47/// It also sets executable permissions on Unix-like systems.
48pub fn download_binary() -> Result<PathBuf, AppError> {
49    let download_url = get_download_url()?;
50    let current_dir = std::env::current_dir().map_err(AppError::IoError)?;
51    let binary_name = crate::utils::platform::get_binary_name();
52    let archive_name = Path::new(&download_url)
53        .file_name()
54        .ok_or_else(|| AppError::CommandFailed("Invalid URL".to_string()))?
55        .to_string_lossy()
56        .to_string();
57    let archive_path = current_dir.join(&archive_name);
58
59    // Download the file with progress bar
60    debug!("Downloading from {}", download_url);
61    let client = Client::builder()
62        .timeout(std::time::Duration::from_secs(300))
63        .build()
64        .map_err(|e| AppError::CommandFailed(format!("Failed to create HTTP client: {}", e)))?;
65    let mut response = client
66        .get(&download_url)
67        .send()
68        .map_err(|e| AppError::CommandFailed(format!("Failed to download: {}", e)))?;
69
70    let total_size = response.content_length().unwrap_or(0);
71    let progress_bar = ProgressBar::new(total_size);
72    progress_bar.set_style(
73        ProgressStyle::default_bar()
74            .template("{spinner:.green} [{elapsed_precise}] [{bar:40.cyan/blue}] {bytes}/{total_bytes} ({eta})")
75            .unwrap()
76            .progress_chars("#>-"));
77
78    let mut file = File::create(&archive_path).map_err(AppError::IoError)?;
79    let mut buffer = [0u8; 8192];
80    let mut downloaded: u64 = 0;
81    loop {
82        let bytes_read = response
83            .read(&mut buffer)
84            .map_err(|e| AppError::CommandFailed(format!("Failed to read response: {}", e)))?;
85        if bytes_read == 0 {
86            break;
87        }
88        file.write_all(&buffer[..bytes_read])
89            .map_err(AppError::IoError)?;
90        downloaded += bytes_read as u64;
91        progress_bar.set_position(downloaded);
92    }
93    progress_bar.finish_with_message("Download complete");
94
95    // Extract the file based on platform
96    debug!("Extracting the binary...");
97    let binary_path = extract_binary(&archive_path, &current_dir, &binary_name)?;
98
99    // Clean up the archive
100    fs::remove_file(&archive_path).ok();
101
102    // Set executable permissions on Unix-like systems
103    if get_platform() != Platform::Windows {
104        Command::new("chmod")
105            .arg("+x")
106            .arg(&binary_path)
107            .output()
108            .map_err(|e| {
109                AppError::CommandFailed(format!("Failed to set executable permission: {}", e))
110            })?;
111    }
112
113    debug!(
114        "StackQL executable successfully installed at: {}",
115        binary_path.display()
116    );
117    Ok(binary_path)
118}

120/// Extracts the StackQL binary from an archive.
121fn extract_binary(
122    archive_path: &Path,
123    dest_dir: &Path,
124    binary_name: &str,
125) -> Result<PathBuf, AppError> {
126    let binary_path = dest_dir.join(binary_name);
127
128    match get_platform() {
129        Platform::MacOS => {
130            // For macOS, we need to use pkgutil
131            let unpacked_dir = dest_dir.join("stackql_unpacked");
132            if unpacked_dir.exists() {
133                fs::remove_dir_all(&unpacked_dir).map_err(AppError::IoError)?;
134            }
135            fs::create_dir_all(&unpacked_dir).map_err(AppError::IoError)?;
136
137            Command::new("pkgutil")
138                .arg("--expand-full")
139                .arg(archive_path)
140                .arg(&unpacked_dir)
141                .output()
142                .map_err(|e| AppError::CommandFailed(format!("Failed to extract pkg: {}", e)))?;
143
144            let extracted_binary = unpacked_dir
145                .join("payload")
146                .join("usr")
147                .join("local")
148                .join("bin")
149                .join("stackql");
150            fs::copy(extracted_binary, &binary_path).map_err(AppError::IoError)?;
151
152            // Clean up
153            fs::remove_dir_all(unpacked_dir).ok();
154        }
155        _ => {
156            // For Windows and Linux, we use the zip file
157            let file = File::open(archive_path).map_err(AppError::IoError)?;
158            let mut archive = ZipArchive::new(file).map_err(|e| {
159                AppError::CommandFailed(format!("Failed to open zip archive: {}", e))
160            })?;
161
162            for i in 0..archive.len() {
163                let mut file = archive.by_index(i).map_err(|e| {
164                    AppError::CommandFailed(format!("Failed to extract file: {}", e))
165                })?;
166
167                let outpath = match file.enclosed_name() {
168                    Some(path) => dest_dir.join(path),
169                    _none => continue,
170                };
171
172                if file.name().ends_with('/') {
173                    fs::create_dir_all(&outpath).map_err(AppError::IoError)?;
174                } else {
175                    let mut outfile = File::create(&outpath).map_err(AppError::IoError)?;
176                    io::copy(&mut file, &mut outfile).map_err(AppError::IoError)?;
177                }
178            }
179
180            // Check if we need to rename the binary on Windows
181            if get_platform() == Platform::Windows {
182                let potential_binary = dest_dir.join("stackql");
183                if potential_binary.exists() && !binary_path.exists() {
184                    fs::rename(potential_binary, &binary_path).map_err(AppError::IoError)?;
185                }
186            }
187        }
188    }
189
190    if !binary_path.exists() {
191        return Err(AppError::CommandFailed(format!(
192            "Binary {} not found after extraction",
193            binary_name
194        )));
195    }
196
197    Ok(binary_path)
198}
