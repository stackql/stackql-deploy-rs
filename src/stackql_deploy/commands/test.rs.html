3//! # Test Command
4//!
5//! Implements the `test` command. Validates that deployed resources are in
6//! the correct desired state.
7//! This is the Rust equivalent of Python's `cmd/test.py` `StackQLTestRunner`.
8
9use std::collections::HashMap;
10use std::time::Instant;
11
12use clap::{Arg, ArgMatches, Command};
13use log::info;
14
15use crate::commands::base::CommandRunner;
16use crate::commands::common_args::{
17    dry_run, env_file, env_var, log_level, on_failure, show_queries, stack_dir, stack_env,
18    FailureAction,
19};
20use crate::core::config::get_resource_type;
21use crate::core::utils::catch_error_and_exit;
22use crate::utils::connection::create_client;
23use crate::utils::display::{print_unicode_box, BorderColor};
24use crate::utils::server::check_and_start_server;
25
26/// Configures the `test` command for the CLI application.
27pub fn command() -> Command {
28    Command::new("test")
29        .about("Run test queries for the stack")
30        .arg(stack_dir())
31        .arg(stack_env())
32        .arg(log_level())
33        .arg(env_file())
34        .arg(env_var())
35        .arg(dry_run())
36        .arg(show_queries())
37        .arg(on_failure())
38        .arg(
39            Arg::new("output-file")
40                .long("output-file")
41                .help("File path to write deployment outputs as JSON")
42                .num_args(1),
43        )
44}
45
46/// Executes the `test` command.
47pub fn execute(matches: &ArgMatches) {
48    let stack_dir_val = matches.get_one::<String>("stack_dir").unwrap();
49    let stack_env_val = matches.get_one::<String>("stack_env").unwrap();
50    let env_file_val = matches.get_one::<String>("env-file").unwrap();
51    let env_vars: Vec<String> = matches
52        .get_many::<String>("env")
53        .map(|v| v.cloned().collect())
54        .unwrap_or_default();
55    let is_dry_run = matches.get_flag("dry-run");
56    let is_show_queries = matches.get_flag("show-queries");
57    let on_failure_val = matches.get_one::<FailureAction>("on-failure").unwrap();
58    let output_file = matches.get_one::<String>("output-file");
59
60    check_and_start_server();
61    let client = create_client();
62    let mut runner = CommandRunner::new(
63        client,
64        stack_dir_val,
65        stack_env_val,
66        env_file_val,
67        &env_vars,
68    );
69
70    let stack_name_display = if runner.stack_name.is_empty() {
71        runner.stack_dir.clone()
72    } else {
73        runner.stack_name.clone()
74    };
75
76    print_unicode_box(
77        &format!(
78            "Testing stack: [{}] in environment: [{}]",
79            stack_name_display, stack_env_val
80        ),
81        BorderColor::Yellow,
82    );
83
84    run_test(
85        &mut runner,
86        is_dry_run,
87        is_show_queries,
88        &format!("{:?}", on_failure_val),
89        output_file.map(|s| s.as_str()),
90    );
91
92    println!("tests complete (dry run: {})", is_dry_run);
93}

95/// Main test workflow matching Python's StackQLTestRunner.run().
96fn run_test(
97    runner: &mut CommandRunner,
98    dry_run: bool,
99    show_queries: bool,
100    _on_failure: &str,
101    output_file: Option<&str>,
102) {
103    let start_time = Instant::now();
104
105    info!(
106        "testing [{}] in [{}] environment {}",
107        runner.stack_name,
108        runner.stack_env,
109        if dry_run { "(dry run)" } else { "" }
110    );
111
112    let resources = runner.manifest.resources.clone();
113
114    for resource in &resources {
115        print_unicode_box(
116            &format!("Processing resource: [{}]", resource.name),
117            BorderColor::Blue,
118        );
119
120        let res_type = get_resource_type(resource).to_string();
121
122        if res_type == "query" {
123            info!("exporting variables for [{}]", resource.name);
124        } else if res_type == "resource" || res_type == "multi" {
125            info!("testing resource [{}], type: {}", resource.name, res_type);
126        } else if res_type == "command" {
127            continue;
128        } else {
129            catch_error_and_exit(&format!("unknown resource type: {}", res_type));
130        }
131
132        let full_context = runner.get_full_context(resource);
133
134        // Get test queries (templates only, not yet rendered)
135        let (test_queries, inline_query) =
136            if let Some(sql_val) = resource.sql.as_ref().filter(|_| res_type == "query") {
137                let iq = runner.render_inline_template(&resource.name, sql_val, &full_context);
138                (HashMap::new(), Some(iq))
139            } else {
140                (runner.get_queries(resource, &full_context), None)
141            };
142
143        // Render statecheck JIT if present
144        let statecheck_rendered = test_queries.get("statecheck").map(|q| {
145            let rendered =
146                runner.render_query(&resource.name, "statecheck", &q.template, &full_context);
147            (rendered, q.options.clone())
148        });
149        let statecheck_retries = test_queries
150            .get("statecheck")
151            .map_or(1, |q| q.options.retries);
152        let statecheck_retry_delay = test_queries
153            .get("statecheck")
154            .map_or(0, |q| q.options.retry_delay);
155
156        // Render exports JIT if present
157        let mut exports_query_str = test_queries
158            .get("exports")
159            .map(|q| runner.render_query(&resource.name, "exports", &q.template, &full_context));
160        let exports_opts = test_queries.get("exports");
161        let exports_retries = exports_opts.map_or(1, |q| q.options.retries);
162        let exports_retry_delay = exports_opts.map_or(0, |q| q.options.retry_delay);
163
164        if res_type == "query" && exports_query_str.is_none() {
165            if let Some(ref iq) = inline_query {
166                exports_query_str = Some(iq.clone());
167            } else {
168                catch_error_and_exit(
169                    "Inline sql must be supplied or an iql file must be present with an 'exports' anchor for query type resources.",
170                );
171            }
172        }
173
174        // Statecheck with optimizations
175        let mut exports_result_from_proxy: Option<Vec<HashMap<String, String>>> = None;
176
177        if res_type == "resource" || res_type == "multi" {
178            let is_correct_state;
179
180            if resource.skip_validation.unwrap_or(false) {
181                info!("Skipping statecheck for {}", resource.name);
182                is_correct_state = true;
183            } else if let Some(ref sq) = statecheck_rendered {
184                is_correct_state = runner.check_if_resource_is_correct_state(
185                    resource,
186                    &sq.0,
187                    sq.1.retries,
188                    sq.1.retry_delay,
189                    dry_run,
190                    show_queries,
191                );
192            } else if let Some(ref eq_str) = exports_query_str {
193                // OPTIMIZATION: Use exports as statecheck proxy
194                info!(
195                    "using exports query as proxy for statecheck test for [{}]",
196                    resource.name
197                );
198                let (state, proxy) = runner.check_state_using_exports_proxy(
199                    resource,
200                    eq_str,
201                    statecheck_retries,
202                    statecheck_retry_delay,
203                    dry_run,
204                    show_queries,
205                );
206                is_correct_state = state;
207                exports_result_from_proxy = proxy;
208            } else {
209                catch_error_and_exit(
210                    "iql file must include either 'statecheck' or 'exports' anchor for validation.",
211                );
212            }
213
214            if !is_correct_state && !dry_run {
215                catch_error_and_exit(&format!("test failed for {}.", resource.name));
216            }
217        }
218
219        // Exports with optimization
220        if let Some(ref eq_str) = exports_query_str {
221            if let Some(ref proxy_result) = exports_result_from_proxy {
222                if res_type == "resource" || res_type == "multi" {
223                    info!(
224                        "reusing exports result from proxy for [{}]...",
225                        resource.name
226                    );
227                    if !resource.exports.is_empty() {
228                        runner.process_exports_from_result(resource, proxy_result);
229                    }
230                }
231            } else {
232                runner.process_exports(
233                    resource,
234                    &full_context,
235                    eq_str,
236                    exports_retries,
237                    exports_retry_delay,
238                    dry_run,
239                    show_queries,
240                    false,
241                );
242            }
243        }
244
245        if res_type == "resource" && !dry_run {
246            info!("test passed for {}", resource.name);
247        }
248    }
249
250    let elapsed = start_time.elapsed();
251    let elapsed_str = format!("{:.2?}", elapsed);
252    info!("test completed in {}", elapsed_str);
253
254    if let Some(of) = output_file {
255        runner.process_stack_exports(dry_run, Some(of), &elapsed_str);
256    }
257}
