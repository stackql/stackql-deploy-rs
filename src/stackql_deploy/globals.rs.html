3//! # Global Configuration Module
4//!
5//! This module provides global variables for the StackQL server configuration.
6//! It manages the global host, port, and connection string settings using `OnceCell` for safe, single initialization.
7//!
8//! ## Features
9//! - Stores global server configuration values (`host`, `port`, `connection_string`) using `OnceCell`.
10//! - Provides initialization functions to set global values (`init_globals`).
11//! - Exposes getter functions for retrieving configured global values from other modules.
12//!
13//! ## Example Usage
14//! ```rust
15//! use crate::globals::{init_globals, server_host, server_port, connection_string};
16//!
17//! fn setup() {
18//!     init_globals("localhost".to_string(), 5444);
19//!     println!("Host: {}", server_host());
20//!     println!("Port: {}", server_port());
21//!     println!("Connection String: {}", connection_string());
22//! }
23//! ```
24
25use once_cell::sync::OnceCell;
26
27use crate::app::{DEFAULT_SERVER_HOST, DEFAULT_SERVER_PORT};
28
29// ============================
30// Global Static Variables
31// ============================
32
33/// Stores the global server host.
34///
35/// The server host is initialized via the `init_globals` function and is only set once per application lifetime.
36static STACKQL_SERVER_HOST: OnceCell<String> = OnceCell::new();
37
38/// Stores the global server port.
39///
40/// The server port is initialized via the `init_globals` function and is only set once per application lifetime.
41static STACKQL_SERVER_PORT: OnceCell<u16> = OnceCell::new();
42
43/// Stores the global connection string used for database connections.
44///
45/// This string is generated using the `init_globals` function based on the provided host and port.
46static STACKQL_CONNECTION_STRING: OnceCell<String> = OnceCell::new();
47
48// ============================
49// Initialization Function
50// ============================
51
52/// Initializes the global variables for host, port, and connection string.
53///
54/// This function must be called once before accessing global values via getter functions.
55/// It uses `OnceCell` to ensure each value is only initialized once.
56///
57/// # Arguments
58/// - `host` - The server host address as a `String`.
59/// - `port` - The server port as a `u16`.
60///
61/// # Example
62/// ```rust
63/// use crate::globals::init_globals;
64/// init_globals("localhost".to_string(), 5444);
65/// ```
66pub fn init_globals(host: String, port: u16) {
67    // Only set if not already set (first initialization wins)
68    STACKQL_SERVER_HOST.set(host.clone()).ok();
69    STACKQL_SERVER_PORT.set(port).ok();
70
71    // Create a connection string and store it globally
72    let connection_string = format!(
73        "host={} port={} user=stackql dbname=stackql application_name=stackql",
74        host, port
75    );
76    STACKQL_CONNECTION_STRING.set(connection_string).ok();
77}
78
79// ============================
80// Getter Functions
81// ============================
82
83/// Retrieves the configured global server host.
84///
85/// If the host is not set via `init_globals`, it returns the default value from `app`.
86///
87/// # Returns
88/// - `&'static str` - The configured server host or the default host.
89///
90/// # Example
91/// ```rust
92/// use crate::globals::{init_globals, server_host};
93/// init_globals("localhost".to_string(), 5444);
94/// assert_eq!(server_host(), "localhost");
95/// ```
96pub fn server_host() -> &'static str {
97    STACKQL_SERVER_HOST
98        .get()
99        .map_or(DEFAULT_SERVER_HOST, |s| s.as_str())
100}
101
102/// Retrieves the configured global server port.
103///
104/// If the port is not set via `init_globals`, it returns the default value from `app`.
105///
106/// # Returns
107/// - `u16` - The configured server port or the default port.
108///
109/// # Example
110/// ```rust
111/// use crate::globals::{init_globals, server_port};
112/// init_globals("localhost".to_string(), 5444);
113/// assert_eq!(server_port(), 5444);
114/// ```
115pub fn server_port() -> u16 {
116    STACKQL_SERVER_PORT
117        .get()
118        .copied()
119        .unwrap_or(DEFAULT_SERVER_PORT)
120}
