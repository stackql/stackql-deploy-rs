19/// Exit with error message. Matches Python's `catch_error_and_exit`.
20pub fn catch_error_and_exit(msg: &str) -> ! {
21    error!("{}", msg);
22    eprintln!("stackql-deploy operation failed");
23    process::exit(1);
24}

26/// Execute a StackQL SELECT query with retry logic.
27/// Returns rows as Vec<HashMap<String, String>>.
28/// Matches Python's `run_stackql_query`.
29pub fn run_stackql_query(
30    query: &str,
31    client: &mut PgwireLite,
32    suppress_errors: bool,
33    retries: u32,
34    delay: u32,
35) -> Vec<HashMap<String, String>> {
36    let mut attempt = 0;
37    let mut last_error: Option<String> = None;
38
39    while attempt <= retries {
40        debug!(
41            "Executing stackql query on attempt {}:\n\n{}\n",
42            attempt + 1,
43            query
44        );
45
46        match execute_query(query, client) {
47            Ok(result) => match result {
48                QueryResult::Data {
49                    columns,
50                    rows,
51                    notices,
52                } => {
53                    // Check for error notices
54                    for notice in &notices {
55                        if notice.contains("error") || notice.starts_with("ERROR") {
56                            last_error = Some(notice.clone());
57                            if !suppress_errors && attempt == retries {
58                                catch_error_and_exit(&format!(
59                                    "Error during stackql query execution:\n\n{}\n",
60                                    notice
61                                ));
62                            }
63                        }
64                    }
65
66                    if rows.is_empty() {
67                        debug!("Stackql query executed successfully, retrieved 0 items.");
68                        if attempt < retries {
69                            thread::sleep(Duration::from_secs(delay as u64));
70                            attempt += 1;
71                            continue;
72                        }
73                        return Vec::new();
74                    }
75
76                    // Convert to Vec<HashMap>
77                    let col_names: Vec<String> = columns.iter().map(|c| c.name.clone()).collect();
78
79                    let result_maps: Vec<HashMap<String, String>> = rows
80                        .iter()
81                        .map(|row| {
82                            let mut map = HashMap::new();
83                            for (i, col_name) in col_names.iter().enumerate() {
84                                let value = row
85                                    .values
86                                    .get(i)
87                                    .cloned()
88                                    .unwrap_or_else(|| "NULL".to_string());
89                                map.insert(col_name.clone(), value);
90                            }
91                            map
92                        })
93                        .collect();
94
95                    // Check for error in results
96                    if !result_maps.is_empty() {
97                        if let Some(err) = result_maps[0].get("error") {
98                            last_error = Some(err.clone());
99                            if !suppress_errors {
100                                if attempt == retries {
101                                    catch_error_and_exit(&format!(
102                                        "Error during stackql query execution:\n\n{}\n",
103                                        err
104                                    ));
105                                } else {
106                                    error!("Attempt {} failed:\n\n{}\n", attempt + 1, err);
107                                }
108                            }
109                            thread::sleep(Duration::from_secs(delay as u64));
110                            attempt += 1;
111                            continue;
112                        }
113
114                        // Check for count query
115                        if let Some(count_str) = result_maps[0].get("count") {
116                            debug!("Stackql query executed successfully, count: {}", count_str);
117                            if let Ok(count) = count_str.parse::<i64>() {
118                                if count > 1 {
119                                    catch_error_and_exit(&format!(
120                                        "Detected more than one resource matching query criteria, expected 0 or 1, got {}",
121                                        count
122                                    ));
123                                }
124                            }
125                            return result_maps;
126                        }
127                    }
128
129                    debug!(
130                        "Stackql query executed successfully, retrieved {} items.",
131                        result_maps.len()
132                    );
133                    return result_maps;
134                }
135                QueryResult::Command(msg) => {
136                    debug!("Command result: {}", msg);
137                    return Vec::new();
138                }
139                QueryResult::Empty => {
140                    debug!("Empty result from query");
141                    if attempt < retries {
142                        thread::sleep(Duration::from_secs(delay as u64));
143                        attempt += 1;
144                        continue;
145                    }
146                    return Vec::new();
147                }
148            },
149            Err(e) => {
150                last_error = Some(e.clone());
151                if attempt == retries {
152                    if !suppress_errors {
153                        catch_error_and_exit(&format!(
154                            "Exception during stackql query execution:\n\n{}\n",
155                            e
156                        ));
157                    }
158                } else {
159                    error!("Exception on attempt {}:\n\n{}\n", attempt + 1, e);
160                }
161            }
162        }
163
164        thread::sleep(Duration::from_secs(delay as u64));
165        attempt += 1;
166    }
167
168    debug!(
169        "All attempts ({}) to execute the query completed.",
170        retries + 1
171    );
172
173    // If suppress_errors and we have an error, return error marker
174    if suppress_errors {
175        if let Some(err) = last_error {
176            let mut error_map = HashMap::new();
177            error_map.insert("_stackql_deploy_error".to_string(), err);
178            return vec![error_map];
179        }
180    }
181
182    Vec::new()
183}

185/// Execute a StackQL DML/DDL command with retry logic.
186/// Matches Python's `run_stackql_command`.
187pub fn run_stackql_command(
188    command: &str,
189    client: &mut PgwireLite,
190    ignore_errors: bool,
191    retries: u32,
192    retry_delay: u32,
193) -> String {
194    let mut attempt = 0;
195
196    // Handle REGISTRY PULL command format
197    let processed_command = if command.starts_with("REGISTRY PULL") {
198        let re = regex::Regex::new(r"(REGISTRY PULL \w+)(::v[\d\.]+)?").unwrap();
199        if let Some(caps) = re.captures(command) {
200            let provider = caps.get(1).map_or("", |m| m.as_str());
201            if let Some(version) = caps.get(2) {
202                format!("{} {}", provider, &version.as_str()[2..])
203            } else {
204                command.to_string()
205            }
206        } else {
207            command.to_string()
208        }
209    } else {
210        command.to_string()
211    };
212
213    while attempt <= retries {
214        debug!(
215            "Executing stackql command (attempt {}):\n\n{}\n",
216            attempt + 1,
217            processed_command
218        );
219
220        match execute_query(&processed_command, client) {
221            Ok(result) => match result {
222                QueryResult::Data { notices, .. } => {
223                    // Check for errors in notices
224                    for notice in &notices {
225                        if error_detected_in_notice(notice) && !ignore_errors {
226                            if attempt < retries {
227                                warn!(
228                                        "Dependent resource(s) may not be ready, retrying in {} seconds (attempt {} of {})...",
229                                        retry_delay, attempt + 1, retries + 1
230                                    );
231                                thread::sleep(Duration::from_secs(retry_delay as u64));
232                                attempt += 1;
233                                continue;
234                            } else {
235                                catch_error_and_exit(&format!(
236                                    "Error during stackql command execution:\n\n{}\n",
237                                    notice
238                                ));
239                            }
240                        }
241                    }
242                    let msg = notices.join("\n");
243                    if !msg.is_empty() {
244                        debug!("Stackql command executed successfully:\n\n{}\n", msg);
245                    }
246                    return msg;
247                }
248                QueryResult::Command(msg) => {
249                    debug!("Stackql command executed successfully:\n\n{}\n", msg);
250                    return msg;
251                }
252                QueryResult::Empty => {
253                    debug!("Command executed with empty result");
254                    return String::new();
255                }
256            },
257            Err(e) => {
258                if !ignore_errors {
259                    if attempt < retries {
260                        warn!(
261                            "Command failed, retrying in {} seconds (attempt {} of {})...",
262                            retry_delay,
263                            attempt + 1,
264                            retries + 1
265                        );
266                        thread::sleep(Duration::from_secs(retry_delay as u64));
267                        attempt += 1;
268                        continue;
269                    }
270                    catch_error_and_exit(&format!(
271                        "Exception during stackql command execution:\n\n{}\n",
272                        e
273                    ));
274                } else {
275                    debug!("Command failed (ignored): {}", e);
276                    return String::new();
277                }
278            }
279        }
280    }
281
282    String::new()
283}

285/// Check if a notice/message indicates an error.
286fn error_detected_in_notice(msg: &str) -> bool {
287    msg.starts_with("http response status code: 4")
288        || msg.starts_with("http response status code: 5")
289        || msg.starts_with("error:")
290        || msg.starts_with("disparity in fields to insert")
291        || msg.starts_with("cannot find matching operation")
292}

294/// Run a test query and check if count == 1 (exists) or count == 0 (deleted).
295/// Matches Python's `run_test`.
296pub fn run_test(
297    resource_name: &str,
298    query: &str,
299    client: &mut PgwireLite,
300    delete_test: bool,
301) -> bool {
302    let result = run_stackql_query(query, client, true, 0, 5);
303
304    if result.is_empty() {
305        if delete_test {
306            debug!("Delete test result true for [{}]", resource_name);
307            return true;
308        } else {
309            debug!("Test result false for [{}]", resource_name);
310            return false;
311        }
312    }
313
314    // Check for error markers
315    if result[0].contains_key("_stackql_deploy_error") || result[0].contains_key("error") {
316        if delete_test {
317            return true;
318        }
319        return false;
320    }
321
322    if let Some(count_str) = result[0].get("count") {
323        if let Ok(count) = count_str.parse::<i64>() {
324            if delete_test {
325                if count == 0 {
326                    debug!("Delete test result true for [{}]", resource_name);
327                    return true;
328                } else {
329                    debug!(
330                        "Delete test result false for [{}], expected 0 got {}",
331                        resource_name, count
332                    );
333                    return false;
334                }
335            } else if count == 1 {
336                debug!("Test result true for [{}]", resource_name);
337                return true;
338            } else {
339                debug!(
340                    "Test result false for [{}], expected 1 got {}",
341                    resource_name, count
342                );
343                return false;
344            }
345        }
346    }
347
348    // If no count field, for non-delete test consider any result as exists
349    if !delete_test && !result.is_empty() {
350        return true;
351    }
352
353    false
354}

356/// Perform retries on a test query.
357/// Matches Python's `perform_retries`.
358pub fn perform_retries(
359    resource_name: &str,
360    query: &str,
361    retries: u32,
362    delay: u32,
363    client: &mut PgwireLite,
364    delete_test: bool,
365) -> bool {
366    let start = Instant::now();
367    let mut attempt = 0;
368
369    while attempt < retries {
370        let result = run_test(resource_name, query, client, delete_test);
371        if result {
372            return true;
373        }
374        let elapsed = start.elapsed().as_secs();
375        info!(
376            "attempt {}/{}: retrying in {} seconds ({} seconds elapsed).",
377            attempt + 1,
378            retries,
379            delay,
380            elapsed
381        );
382        thread::sleep(Duration::from_secs(delay as u64));
383        attempt += 1;
384    }
385
386    false
387}

396/// Pull providers using the StackQL server.
397/// Matches Python's `pull_providers`.
398pub fn pull_providers(providers: &[String], client: &mut PgwireLite) {
399    let installed = run_stackql_query("SHOW PROVIDERS", client, false, 0, 5);
400
401    for provider in providers {
402        if provider.contains("::") {
403            // Versioned provider
404            let parts: Vec<&str> = provider.splitn(2, "::").collect();
405            let name = parts[0];
406            let version = parts[1];
407
408            let found = installed.iter().any(|p| {
409                p.get("name").is_some_and(|n| n == name)
410                    && p.get("version").is_some_and(|v| v == version)
411            });
412
413            if found {
414                info!("Provider '{}' is already installed.", provider);
415            } else {
416                // Check if a higher version is installed
417                let higher_installed = installed.iter().any(|p| {
418                    p.get("name").is_some_and(|n| n == name)
419                        && p.get("version")
420                            .is_some_and(|v| is_version_higher(v, version))
421                });
422
423                if higher_installed {
424                    info!(
425                        "Provider '{}' - a higher version is already installed.",
426                        provider
427                    );
428                } else {
429                    info!("Pulling provider '{}'...", provider);
430                    let cmd = format!("REGISTRY PULL {}", provider);
431                    let msg = run_stackql_command(&cmd, client, false, 0, 5);
432                    if !msg.is_empty() {
433                        info!("{}", msg);
434                    }
435                }
436            }
437        } else {
438            let found = installed.iter().any(|p| p.get("name") == Some(provider));
439
440            if found {
441                info!("Provider '{}' is already installed.", provider);
442            } else {
443                info!("Pulling provider '{}'...", provider);
444                let cmd = format!("REGISTRY PULL {}", provider);
445                let msg = run_stackql_command(&cmd, client, false, 0, 5);
446                if !msg.is_empty() {
447                    info!("{}", msg);
448                }
449            }
450        }
451    }
452}

454/// Compare version strings. Returns true if installed > requested.
455fn is_version_higher(installed: &str, requested: &str) -> bool {
456    let parse = |v: &str| -> u64 { v.replace(['v', '.'], "").parse::<u64>().unwrap_or(0) };
457    parse(installed) > parse(requested)
458}

460/// Update global context with exported values.
461///
462/// Each export is stored under two keys:
463///
464/// - **`{key}`** — the global (unscoped) key.  This can be overridden by a
465///   subsequent resource that exports a variable with the same name, so it
466///   always reflects the *most recent* export value.
467///
468/// - **`{resource_name}.{key}`** — the resource-scoped (fully qualified) key.
469///   This is written **once** and never overwritten, so it is immutable once
470///   set.  Consumers that need an unambiguous reference should use this form.
471///
472/// Matches Python's `export_vars`.
473pub fn export_vars(
474    global_context: &mut HashMap<String, String>,
475    resource_name: &str,
476    export_data: &HashMap<String, String>,
477    protected_exports: &[String],
478) {
479    for (key, value) in export_data {
480        let is_protected = protected_exports.contains(key);
481        let display_value = if is_protected {
482            "*".repeat(value.len())
483        } else {
484            value.clone()
485        };
486
487        // --- resource-scoped key (immutable: only written if not already set) ---
488        let scoped_key = format!("{}.{}", resource_name, key);
489        global_context.entry(scoped_key.clone()).or_insert_with(|| {
490            info!(
491                "set {} [{}] to [{}] in exports",
492                if is_protected {
493                    "protected variable"
494                } else {
495                    "variable"
496                },
497                scoped_key,
498                display_value,
499            );
500            value.clone()
501        });
502
503        // --- global (unscoped) key (can be overridden by later resources) ---
504        info!(
505            "set {} [{}] to [{}] in exports",
506            if is_protected {
507                "protected variable"
508            } else {
509                "variable"
510            },
511            key,
512            display_value,
513        );
514        global_context.insert(key.clone(), value.clone());
515    }
516}

518/// Check if exports result can serve as a statecheck proxy.
519/// Returns true if result is non-empty and has no errors.
520/// Matches Python's `check_exports_as_statecheck_proxy`.
521pub fn check_exports_as_statecheck_proxy(exports_result: &[HashMap<String, String>]) -> bool {
522    debug!(
523        "Checking exports result as statecheck proxy: {} rows",
524        exports_result.len()
525    );
526
527    if exports_result.is_empty() {
528        debug!("Empty exports result, treating as statecheck failure");
529        return false;
530    }
531
532    // Check for error conditions
533    if exports_result[0].contains_key("_stackql_deploy_error") {
534        debug!("Error in exports result, treating as statecheck failure");
535        return false;
536    }
537    if exports_result[0].contains_key("error") {
538        debug!("Error in exports result, treating as statecheck failure");
539        return false;
540    }
541
542    debug!("Valid exports result, treating as statecheck success");
543    true
544}

556/// Run an external script and capture output.
557/// Matches Python's `run_ext_script`.
558pub fn run_ext_script(
559    cmd: &str,
560    expected_exports: Option<&[String]>,
561) -> Option<HashMap<String, String>> {
562    debug!("Running external script: {}", cmd);
563
564    let output = match std::process::Command::new("sh").arg("-c").arg(cmd).output() {
565        Ok(output) => output,
566        Err(e) => {
567            catch_error_and_exit(&format!("Script failed: {}", e));
568        }
569    };
570
571    let stdout = String::from_utf8_lossy(&output.stdout).to_string();
572    debug!("Script output: {}", stdout);
573
574    if !output.status.success() {
575        let stderr = String::from_utf8_lossy(&output.stderr);
576        catch_error_and_exit(&format!(
577            "Script failed with status {:?}: {}",
578            output.status.code(),
579            stderr
580        ));
581    }
582
583    match expected_exports {
584        Some(exports) if !exports.is_empty() => {
585            match serde_json::from_str::<HashMap<String, String>>(&stdout) {
586                Ok(exported_vars) => {
587                    for export in exports {
588                        if !exported_vars.contains_key(export) {
589                            catch_error_and_exit(&format!(
590                                "Exported variable '{}' not found in script output",
591                                export
592                            ));
593                        }
594                    }
595                    Some(exported_vars)
596                }
597                Err(_) => {
598                    catch_error_and_exit(&format!(
599                        "External scripts must return valid JSON: {}",
600                        stdout
601                    ));
602                }
603            }
604        }
605        _ => None,
606    }
607}

619/// Execute a DML command (INSERT / UPDATE / DELETE), optionally capturing
620/// the `RETURNING *` result as the first row.
621///
622/// Returns `(command_message, Option<first_row>)`.  When the DML includes
623/// `RETURNING *` and the provider returns rows, the first row is captured.
624/// If no rows are returned (provider returned no body), `None` is returned –
625/// this is **not** an error.
626pub fn run_stackql_dml_returning(
627    command: &str,
628    client: &mut PgwireLite,
629    ignore_errors: bool,
630    retries: u32,
631    retry_delay: u32,
632) -> (String, Option<HashMap<String, String>>) {
633    let mut attempt = 0u32;
634
635    while attempt <= retries {
636        debug!(
637            "Executing stackql DML (attempt {}):\n\n{}\n",
638            attempt + 1,
639            command
640        );
641
642        match execute_query(command, client) {
643            Ok(result) => match result {
644                QueryResult::Data {
645                    columns,
646                    rows,
647                    notices,
648                } => {
649                    // Check for errors in notices before accepting the result.
650                    let mut error_noticed = false;
651                    for notice in &notices {
652                        if error_detected_in_notice(notice) && !ignore_errors {
653                            if attempt < retries {
654                                warn!(
655                                    "DML error in notice, retrying in {} seconds (attempt {} of {})...",
656                                    retry_delay, attempt + 1, retries + 1
657                                );
658                                thread::sleep(Duration::from_secs(retry_delay as u64));
659                                attempt += 1;
660                                error_noticed = true;
661                                break;
662                            } else {
663                                catch_error_and_exit(&format!(
664                                    "Error during stackql DML execution:\n\n{}\n",
665                                    notice
666                                ));
667                            }
668                        }
669                    }
670                    if error_noticed {
671                        continue;
672                    }
673
674                    // Capture RETURNING * first row (if any).
675                    let first_row = if !rows.is_empty() {
676                        let col_names: Vec<String> =
677                            columns.iter().map(|c| c.name.clone()).collect();
678                        let row = &rows[0];
679                        let mut map = HashMap::new();
680                        for (i, col_name) in col_names.iter().enumerate() {
681                            let value = row.values.get(i).cloned().unwrap_or_default();
682                            map.insert(col_name.clone(), value);
683                        }
684                        Some(map)
685                    } else {
686                        None
687                    };
688
689                    let msg = notices.join("\n");
690                    return (msg, first_row);
691                }
692                QueryResult::Command(msg) => {
693                    return (msg, None);
694                }
695                QueryResult::Empty => {
696                    return (String::new(), None);
697                }
698            },
699            Err(e) => {
700                if !ignore_errors {
701                    if attempt < retries {
702                        warn!(
703                            "DML failed, retrying in {} seconds (attempt {} of {})...",
704                            retry_delay,
705                            attempt + 1,
706                            retries + 1
707                        );
708                        thread::sleep(Duration::from_secs(retry_delay as u64));
709                        attempt += 1;
710                        continue;
711                    }
712                    catch_error_and_exit(&format!(
713                        "Exception during stackql DML execution:\n\n{}\n",
714                        e
715                    ));
716                } else {
717                    debug!("DML failed (ignored): {}", e);
718                    return (String::new(), None);
719                }
720            }
721        }
722    }
723
724    (String::new(), None)
725}

727/// Flatten a single RETURNING * row into dotted context keys and insert them
728/// into `context`.
729///
730/// For each column `col` in `row`:
731/// - `callback.{col}` is set (shorthand for the current resource's own `.iql`
732///   templates).
733/// - `{resource_name}.callback.{col}` is set (fully-qualified key accessible
734///   by downstream resources).
735///
736/// If a column value is a JSON object it is recursively expanded:
737/// `"ProgressEvent" = {"OperationStatus":"SUCCESS","RequestToken":"abc"}`
738/// produces:
739/// ```text
740/// callback.ProgressEvent.OperationStatus = SUCCESS
741/// callback.ProgressEvent.RequestToken    = abc
742/// ```
743pub fn flatten_returning_row(
744    row: &HashMap<String, String>,
745    resource_name: &str,
746    context: &mut HashMap<String, String>,
747) {
748    for (col, val) in row {
749        let short_prefix = format!("callback.{}", col);
750        let full_prefix = format!("{}.callback.{}", resource_name, col);
751        flatten_value_into_context(&short_prefix, &full_prefix, val, context);
752    }
753}
754
755/// Recursively expand a string value (possibly JSON) into dotted context keys.
756fn flatten_value_into_context(
757    short_prefix: &str,
758    full_prefix: &str,
759    value: &str,
760    context: &mut HashMap<String, String>,
761) {
762    if let Ok(json) = serde_json::from_str::<serde_json::Value>(value) {
763        if json.is_object() {
764            flatten_json_into_context(short_prefix, full_prefix, &json, context);
765            return;
766        }
767    }
768    context.insert(short_prefix.to_string(), value.to_string());
769    context.insert(full_prefix.to_string(), value.to_string());
770}
771
772fn flatten_json_into_context(
773    short_prefix: &str,
774    full_prefix: &str,
775    value: &serde_json::Value,
776    context: &mut HashMap<String, String>,
777) {
778    match value {
779        serde_json::Value::Object(map) => {
780            for (k, v) in map {
781                let new_short = format!("{}.{}", short_prefix, k);
782                let new_full = format!("{}.{}", full_prefix, k);
783                flatten_json_into_context(&new_short, &new_full, v, context);
784            }
785        }
786        serde_json::Value::String(s) => {
787            context.insert(short_prefix.to_string(), s.clone());
788            context.insert(full_prefix.to_string(), s.clone());
789        }
790        other => {
791            let s = other.to_string();
792            context.insert(short_prefix.to_string(), s.clone());
793            context.insert(full_prefix.to_string(), s);
794        }
795    }
796}

798/// Check whether a short-circuit condition is met using already-captured
799/// callback data.
800///
801/// `field` is a dot-path into the captured result (e.g.
802/// `"ProgressEvent.OperationStatus"`), looked up as `callback.{field}` in
803/// `context`.  Returns `true` if the value equals `expected_value`.
804/// Returns `false` (no short-circuit) if the field is absent.
805pub fn check_short_circuit(
806    context: &HashMap<String, String>,
807    field: &str,
808    expected_value: &str,
809) -> bool {
810    let lookup_key = format!("callback.{}", field);
811    match context.get(&lookup_key) {
812        Some(val) => {
813            let result = val == expected_value;
814            if result {
815                info!(
816                    "short-circuit condition met: {} = {} (skipping callback poll)",
817                    lookup_key, expected_value
818                );
819            }
820            result
821        }
822        None => {
823            debug!(
824                "short-circuit field '{}' not found in context, proceeding with callback poll",
825                lookup_key
826            );
827            false
828        }
829    }
830}

832/// Poll a callback query until the `success` (or `count`) column returns a
833/// truthy value, or `retries` are exhausted.
834///
835/// Returns `true` on success, `false` when retries are exhausted (the caller
836/// is responsible for treating exhaustion as an error).
837pub fn run_callback_poll(
838    resource_name: &str,
839    query: &str,
840    retries: u32,
841    retry_delay: u32,
842    client: &mut PgwireLite,
843) -> bool {
844    let mut attempt = 0u32;
845
846    while attempt <= retries {
847        debug!(
848            "Callback poll for [{}] attempt {}:\n\n{}\n",
849            resource_name,
850            attempt + 1,
851            query
852        );
853
854        let result = run_stackql_query(query, client, true, 0, 0);
855
856        if !result.is_empty() {
857            let row = &result[0];
858
859            // Check `success` column (primary).
860            if let Some(success_val) = row.get("success") {
861                if success_val == "1" || success_val.to_lowercase() == "true" {
862                    info!(
863                        "[{}] callback poll succeeded on attempt {}",
864                        resource_name,
865                        attempt + 1
866                    );
867                    return true;
868                }
869            }
870
871            // Check `count` column (alternative).
872            if let Some(count_val) = row.get("count") {
873                if count_val == "1" {
874                    info!(
875                        "[{}] callback poll succeeded (count=1) on attempt {}",
876                        resource_name,
877                        attempt + 1
878                    );
879                    return true;
880                }
881            }
882        }
883
884        if attempt < retries {
885            info!(
886                "[{}] callback poll attempt {}/{}: retrying in {} seconds...",
887                resource_name,
888                attempt + 1,
889                retries + 1,
890                retry_delay
891            );
892            thread::sleep(Duration::from_secs(retry_delay as u64));
893        }
894        attempt += 1;
895    }
896
897    false
898}
